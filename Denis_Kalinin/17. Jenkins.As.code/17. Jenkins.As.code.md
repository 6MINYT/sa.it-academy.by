## Script
```
pipeline {
    agent { label 'vagrant'}
    triggers { 
        cron('H 02 * * 7')
    }
    stages {
        
        stage('Connection:') {
            steps {
                echo 'ping mail.ru..'
                sh """
                ping mail.ru -c 3
                """
            }
        }
        
        stage('Interfaces:') {
        steps {
                echo 'Interfaces:'
                sh """
                /usr/sbin/ip a
                """
            }
        }
        
        stage('Nmap scan:') {
            steps {
                echo 'installing nmap & scaning..'
                sh """
                sudo /bin/yum install nmap -y
                date > report.txt
                echo -e "\\n Scan nmap: \\n" >> report.txt
                nmap -sP 192.168.100.1/24 >> report.txt
                cat report.txt
                """
            }
        }
        
        stage('Speedtest scan:') {
            steps {
                echo 'Speedtest:'
                sh """
                sudo pip install speedtest-cli
                echo -e "\\n Scan speedtest: \\n" >> report.txt
                speedtest-cli >> report.txt
                """
            }
        }
        
        stage('Show report:') {
            steps {
                echo 'Speedtest:'
                sh """
                cat report.txt
                """
            }
        }
        
        stage('Push to git:') {
            steps {
                echo 'Speedtest:'
                sh """
                git add report.txt
                git commit report.txt -m "report from jenkins"
                git push -u origin main
                """
            }
        }
        
        stage('Remove all:') {
            steps {
                echo 'removing:'
                sh """
                sudo /bin/yum remove nmap -y
                rpm -qa | grep speedtest | xargs -I {} sudo /bin/yum -y remove {}
                """
            }
        }
        
        
    }
}
```
