  ### Script
  
  ```bash
  
 #!/bin/bash

 my_path=/home/yunga/first_repo

 cd $my_path

 for every_remote_repo in $(git remote); do
    date >> run.log
    git push -u $every_remote_repo --all
    echo "pull in $every_remote_repo" >> run.log
    done

  ```
