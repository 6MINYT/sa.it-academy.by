      name: alertmanager

      - name: alertmanager

          - "--config.file=/etc/alertmanager/config.yml"
          - "--storage.path=/alertmanager"
        ports:
        - name: alertmanager

              cpu: 500m
              memory: 500M
            limits:

              memory: 1Gi

        - name: templates-volume
          mountPath: /etc/alertmanager-templates

          mountPath: /alertmanager

      - name: config-volume
        configMap:
          name: alertmanager-config
      - name: templates-volume
        configMap:
          name: alertmanager-templates
      - name: alertmanager
